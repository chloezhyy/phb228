loop_zscore <- function(x) {
  n <- length(x)
  res <- numeric(n)
  m <- mean(x, na.rm = TRUE)
  s <- sd(x, na.rm = TRUE)
  for (i in 1:n) {
    res[i] <- (x[i] - m) / s
  }
  return(res)
}

The Frobenius norm of the difference represents the “energy” or variance that is discarded by ignoring the second and third singular values. The Rosenbrock function is defined as: \[f(x, y) = (1 - x)^2 + 100(y - x^2)^2\] It is a classic test problem for optimization algorithms because the global minimum lies in a narrow, parabolic valley.

(a)

rosenbrock <- function(x) {

  (1 - x[1])^2 + 100 * (x[2] - x[1]^2)^2
}

(b)

\(\frac{\partial f}{\partial x} = -2(1 - x) - 400x(y - x^2)\) \(\frac{\partial f}{\partial y} = 200(y - x^2)\)

rosenbrock_grad <- function(x) {
  df_dx <- -2 * (1 - x[1]) - 400 * x[1] * (x[2] - x[1]^2)
  df_dy <- 200 * (x[2] - x[1]^2)
  return(c(df_dx, df_dy))
}

(c)

This implementation performs iterative updates using the rule \(x_{new} = x_{old} - \eta \cdot \nabla f(x)\), where \(\eta\) is the learning rate. We use the Euclidean norm of the gradient as a convergence metric to stop the search once the gradient is close to zero, ensuring that the algorithm identifies the local minimum efficiently without unnecessary computation

gradient_descent <- function(init, grad_fn, learning_rate, max_iter, tol) {
  curr_x <- init
  for (i in 1:max_iter) {
    grad <- grad_fn(curr_x)
    

    grad_norm <- sqrt(sum(grad^2))
    

    if (grad_norm < tol) {
      return(list(solution = curr_x, 
                  value = rosenbrock(curr_x), 
                  iterations = i))
    }
    

    curr_x <- curr_x - learning_rate * grad
  }
  
  return(list(solution = curr_x, 
              value = rosenbrock(curr_x), 
              iterations = max_iter))
}

The global minimum of the Rosenbrock function is at \((1, 1)\), where \(f(1, 1) = 0\). Using the gradient descent method, we converged to a solution very close to \((1, 1)\). Gradient descent can be slow on this function because the “valley” is very flat, but with a sufficient number of iterations and a proper learning rate, it successfully identifies the minimum.

run_simulation <- function(n_samples, n_reps, dist_type) {

  results <- replicate(n_reps, {

    if (dist_type == "normal") {
      data <- rnorm(n_samples, mean = 0, sd = 1)
    } else if (dist_type == "t3") {
      data <- rt(n_samples, df = 3)
    } else if (dist_type == "contaminated") {
      # 90% from N(0,1), 10% from N(0,10)
      indices <- sample(c(1, 2), size = n_samples, replace = TRUE, prob = c(0.9, 0.1))
      data <- numeric(n_samples)
      data[indices == 1] <- rnorm(sum(indices == 1), 0, 1)
      data[indices == 2] <- rnorm(sum(indices == 2), 0, 10)
    }
    

    c(mean = mean(data),
      median = median(data),
      trimmed = mean(data, trim = 0.1))
  })
  

  return(as.data.frame(t(results)))
}

We implement a function that performs nonparametric bootstrapping to estimate the standard error and three types of 95% confidence intervals: Percentile, Basic (Reverse Percentile), and Normal.

bootstrap_mean <- function(data, n_boot = 1000, conf_level = 0.95) {
  set.seed(42) 
  n <- length(data)
  orig_mean <- mean(data)
  

  boot_means <- replicate(n_boot, {
    resample <- sample(data, size = n, replace = TRUE)
    mean(resample)
  })
  

  se_boot <- sd(boot_means)
  

  alpha <- 1 - conf_level
  

  ci_percentile <- quantile(boot_means, probs = c(alpha/2, 1 - alpha/2))
  
 
  z_crit <- qnorm(1 - alpha/2)
  ci_normal <- c(orig_mean - z_crit * se_boot, orig_mean + z_crit * se_boot)
  

  # Formula: [2*theta_hat - theta_*(1-alpha/2), 2*theta_hat - theta_*(alpha/2)]
  qs <- quantile(boot_means, probs = c(alpha/2, 1 - alpha/2))
  ci_basic <- c(2 * orig_mean - qs[2], 2 * orig_mean - qs[1])
  
  return(list(
    orig_mean = orig_mean,
    se = se_boot,
    boot_dist = boot_means,
    intervals = list(
      percentile = ci_percentile,
      normal = ci_normal,
      basic = ci_basic
    )
  ))
}

the null hypothesis \(H_0: \mu_{treatment} = \mu_{control}\) by shifting the groups to have the same mean, thereby simulating the null distribution.

bootstrap_ttest <- function(control, treatment, n_boot = 2000) {
  set.seed(123)
  obs_diff <- mean(treatment) - mean(control)
  

  combined_mean <- mean(c(control, treatment))
  control_null <- control - mean(control) + combined_mean
  treatment_null <- treatment - mean(treatment) + combined_mean
  

  null_dist <- replicate(n_boot, {
    boot_c <- sample(control_null, replace = TRUE)
    boot_t <- sample(treatment_null, replace = TRUE)
    mean(boot_t) - mean(boot_c)
  })
  

  p_val <- mean(abs(null_dist) >= abs(obs_diff))
  
  return(list(obs_diff = obs_diff, null_dist = null_dist, p_value = p_val))
}
